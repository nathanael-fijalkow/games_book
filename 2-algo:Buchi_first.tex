\begin{algorithm}[H]

 \KwData{A B&uuml;chi game.}
 \SetKwFunction{FSolve}{Solve}
 \SetKwProg{Fn}{Function}{:}{}
 \DontPrintSemicolon

\Fn{\FSolve{$\Game$}}{
	$X \leftarrow \AttrE(\Win)$

	\If{$X = V$}{
		\Return{$V$}
	}
	\Else{
		Let $\Game'$ the subgame of $\Game$ induced by $V \setminus \AttrA(X)$
		
		\Return{$\FSolve{$\Game'$}$}
	}
}

\end{algorithm}
