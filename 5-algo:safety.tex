\begin{algorithm}[H]

	\KwData{An MDP $ \mdp $, a colour $\genColour$}
	\SetKwFunction{KwOPAS}{$\OPAS$}
	\SetKwProg{Fn}{Function}{:}{}
	
		$Y \leftarrow \vertices \setminus\vertices_{\genColour}$ \;
		\Repeat{$Y' \neq Y$}{
			$Y' \leftarrow Y$\;
			$Y \leftarrow \safeOP(Y)$
		}
		\Return{$Y$} \tcp*{$Y$ is now a closed set}


\end{algorithm}
