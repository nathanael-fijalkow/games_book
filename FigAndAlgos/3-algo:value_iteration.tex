 \KwData{A parity game with $n$ vertices priorities in $[1,d]$ and a $(n,d/2)$-universal tree $T$.}
 \DontPrintSemicolon

\For{$v \in V$}{
$\mu(v) \leftarrow \ell_{\max}$
}
     
\Repeat{$\forall v \in V,\ \mu \le \Op_v(\mu)$}{
Choose $v \in V$ which is neglected

$\mu \leftarrow \min(\mu, \Op_v(\mu))$}

\Return{$\mu$}
