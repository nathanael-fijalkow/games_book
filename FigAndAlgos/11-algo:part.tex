\begin{algorithm}[H]

 \KwData{A "vector addition system with states"
   $\mathcal{V}=(\Loc,\Act,\dd)$, $M\in 2^\Act$, $F\subseteq\Act$}

$\Act'\leftarrow\emptyset$;

\ForEach{$a\in\Act$}{\If{$\mathcal{V}$ has a "non-negative" "multi-cycle"
    "suitable"
    for $(M\cup\{\{a\}\},F)$}{$\Act'\leftarrow\Act'\cup\{a\}$}}

$\mathcal{V}'\leftarrow \textrm{subsystem induced by $\Act'$}$ ;

\Return{$\mathrm{SCC}(\mathcal{V}')$}

\end{algorithm}
