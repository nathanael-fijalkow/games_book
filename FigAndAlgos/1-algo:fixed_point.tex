\begin{algorithm}[H]

 \KwData{$(X,\le)$ a finite lattice and $\Op_i : X \to X$ a set of monotonic operators for $i \in I$}

$x \leftarrow \bot$ 

\Repeat{$x$ is a fixed point of $(\Op_i)_{i \in I}$}{
	Choose $i \in I$ such that $\neg (\Op_i(x) \le x)$

	$x \leftarrow \max(x, \Op_i(x))$ 
}

\Return{$x$}

\end{algorithm}
