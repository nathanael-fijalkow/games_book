\begin{algorithm}[H]

 \KwData{A B{\"u}chi game.}

$Y_0 \leftarrow V$ 

$k \leftarrow 0$ 
     
\Repeat{$Y_{k+1} = Y_k$}{
	$Y_{k+1} \leftarrow Y_k \cap \AttrE \left( \Win \cap \PreE(Y_k) \right)$ 

	$k \leftarrow k + 1$
}

\Return{$Y_k$}

\end{algorithm}
