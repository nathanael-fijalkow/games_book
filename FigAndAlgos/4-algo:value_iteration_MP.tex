 \KwData{A mean payoff game}

\For{$v \in V$}{
$\mu(v) \leftarrow 0$
}
     
\lRepeat{$\mu = \Lift(\mu)$}{
$\mu \leftarrow \Lift(\mu)$}

\Return{$\{v\in V\mid \mu(v)\neq -\infty\}$}
